427  template<typename T>
428  inline void Handler(const T retType, int pin = 0, int low = 0, int high = 0,
429  int pdd = 0, const char *what = "",
430  const char *type = "") {
431  int theRet = static_cast<int>(retType); //Handle it to an int
432  //(Some return floats, etc... just easier on this end)
433 
434  //if(strlen(type) > 0 && strlen(what) > 0) printf("%s: %s\n", what, type); //If what and type aren't blank, print them
435 
436  switch(theRet) {
437  case NEO_PIN_ERROR:
438  throw PinError(pin, low, high);
439  case NEO_PERIOD_ERROR:
440  throw PeriodError(pdd);
441  case NEO_UNUSABLE_ERROR:
442  case NEO_UNUSABLE_EXPORT_ERROR:
443  throw UnusableError(what, pin);
444  case NEO_DUTY_ERROR:
445  throw DutyError(pdd, 0, 255); //Default duty low is 0 and high is 255
446  case NEO_EXPORT_ERROR:
447  throw InitError(what);
448  case NEO_DIR_ERROR:
449  throw DirError(pdd);
450  case NEO_SCALE_ERROR:
451  case NEO_READ_ERROR:
452  throw ReadWriteError(what, type);
453  default: break;
454  }
455  }
